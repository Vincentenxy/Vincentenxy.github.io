Quicksort

快速排序

来说一下快速排序的基本思路吧！

对于一个数组 list [ p;r ],要进行一般一下三个步骤：

（1）分解：首先将数组以一个元素为基准划分为三段，小于基准元素的为一段，基准元素为一段，大于基准元素的为一段。

（2）递归求解：通过递归的方法分别为上面的三段数列进行排序。

（3）：合并：有序排序是就地进行的，所以在大于、小于基准元素的排好序之后就不用再执行任何计算了，已经排好序了。

下面看一下算法实现代码：

头文件：

template < class Type >

int Partition(Type a[], int p, int r)

{

int i = p, j = r + 1 ;

int temp;

Type x = a[p];

while (i < j)

{

     while (a[++i] < x&& i < r);

     while (a[--j] > x);

     if (i < j)

     {

         temp= a[i];

         a[i] = a[j];

         a[j] = temp;

     }

}

a[p] = a[j];

a[j] = x;

return j;

}

template

void QuickSort(Type a[], int p, int r)

{

if (p < r)

{

     int q = Partition(a, p, r);

     QuickSort(a, p, q - 1);

     QuickSort(a, q + 1, r);

}

}

main.cpp文件

template < class Type>

void MergeSort( Type a[], int n)

{

Type *b = new Type[n];

int s = 1;

while (s < n)

{

     MergePass(a, b, s, n);

     s+= s;

     MergePass(b,a, s, n);

     s+= s;

}

}

template < class Type>

void MergePass( Type x[], Type y[], int s, int n)

{

int i = 0;



while ( i <= n - 2 * s )

{

     Merge(x, y, i, i + s - 1, i + 2 * s - 1);

     i= i + 2 * s;

}

if (i + s < n)

     Merge(x, y, i, i + s - 1, n - 1);

else

     for (int j = i; j <= n - 1; j++)

         y[j] = x[j];

}

template

void Merge(Type c[], Type d[], int a, int m, int r)

{

int i = a, j = m + 1, k = a;



while ((i <= m) && (j<= r))

{

     if (c[i] <= c[j])

         d[k++] = c[i++];

     else

         d[k++] = c[j++];

}

if (i > m)

     for (int q = j; q <= r; q++)

         d[k++] = c[q];

else

     for(int q = i; q <= m; q ++)

         d[k++] = c[q];

}
